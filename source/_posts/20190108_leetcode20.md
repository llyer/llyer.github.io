问题连接：[20. 有效的括号](https://leetcode-cn.com/problems/valid-parentheses/)

```java
class Solution {
    /**
     *
     * 这道题完美契合栈的所有特点，使用栈可以直接匹配解决
     *
     * @param s
     * @return
     */
    public static boolean isValid(String s) {
        Stack<Character> stack = new Stack();
        char[] chars = s.toCharArray();
        for (char c: chars) {
            if (stack.size() > 0) {
                if (c == '}' && stack.peek() == '{') {
                    stack.pop();
                    continue;
                }
                if (c == ']' && stack.peek() == '[') {
                    stack.pop();
                    continue;
                }
                if (c == ')' && stack.peek() == '(') {
                    stack.pop();
                    continue;
                }
            }
            stack.push(c);
        }
        return stack.size() == 0 ? true : false;
    }
}
```
